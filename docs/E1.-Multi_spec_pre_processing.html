8. Generating the VI-s

multi$NDVI<- (multi$nir - multi$red) / (multi$red + multi$nir)
multi$NDRE<- (multi$nir - multi$red_edge) / (multi$red_edge + multi$nir)
multi$EVI <- 2.5 * (multi$nir - multi$red) / (multi$nir + 6 * multi$red - 7.5 * multi$blue + 1)
multi$GDVI <- multi$nir - multi$green
multi$GNDVI <- (multi$nir - multi$green) / (multi$nir + multi$green)
multi$NRI <- multi$red / (multi$nir + multi$red_edge + multi$red)
multi$SAVI <- 1.5 * ((multi$nir - multi$red) / (multi$nir + multi$red + 0.5))
multi$ENDVI <- (multi$nir + multi$green - 2 * multi$blue) / (multi$nir + multi$green + 2 * multi$blue)
multi$GIPVI <- multi$nir / (multi$nir + multi$green)
multi$PSRI <- (multi$red - multi$blue)/ multi$red_edge
