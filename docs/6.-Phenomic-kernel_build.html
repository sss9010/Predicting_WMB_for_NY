2. Creating a relationship matrix for each list item

# Initialize output list
MT_sim_mat_pr <- list()

# Loop through each environment × timepoint entry in the list
for (name in names(MT_pr_w)) {
  mat <- as.matrix(MT_pr_w[[name]])
  
  # Euclidean distance matrix
  eu_dist <- dist(mat, method = "euclidean")
  
  # Convert to full matrix and compute similarity
  sim_mat <- 1 / (1 + as.matrix(eu_dist))
  
  # Preserve GID as row and column names
  rownames(sim_mat) <- rownames(mat)
  colnames(sim_mat) <- rownames(mat)
  
  # Save to output list
  MT_sim_mat_pr[[name]] <- sim_mat
}
