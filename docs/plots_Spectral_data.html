#Select and index to plot
index<- "NDVI"
# Get unique levels of Env
env_levels <- unique(multi$Env)

# Initialize an empty list to store results
results_list <- list()

# Outer loop for Env levels
for (env in env_levels) {
  # Filter data for the current environment
  multi_env <- multi %>% filter(Env == env) %>% arrange(JD)
  multi_env$GID<- as.factor(multi_env$GID)
  
  # Get unique dates for the current environment
  dates <- unique(multi_env$JDf)
  
  # Inner loop for dates
  for (i in dates) {
    # Filter data for the current date
    multi_date <- subset(multi_env, JDf == i)
    
    # Perform the analysis
    result <- ASRtriala::audit.single(data = multi_date, gen = "GID", row = "Row", col = "Column", resp = index)
    
    # Store the result in the list with a descriptive name
    results_list[[paste0("Env_", env, "_JD_", i, "_", index)]] <- result
  }
}

# Printing the plots
for (name in names(results_list)) {
  # Access and display the trial plot for the current result
  print(results_list[[name]]$trial.plot)
  #print(results_list[[name]]$trial.plot$data$Env[1])
  print(results_list[[name]]$trial.plot$data$timepoint[1])
  #print(results_list[[name]]$trial.plot$data$JDf[1])
  # Optionally, print a message to the console for tracking
  cat(name, "\n")
}
