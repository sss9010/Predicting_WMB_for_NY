calculate_h2_mq_single <- function(data, env, trait, A_matrix) {
  
  cat("\n========================================\n")
  cat("Environment:", env, "| Trait:", trait, "\n")
  cat("========================================\n")
  
  # Filter data
  data_subset <- data %>%
    filter(Env == env) %>%
    filter(!is.na(!!sym(trait))) %>%
    filter(!!sym(trait) > 0) %>%  # Remove zeros/negatives if needed
    filter(!is.na(Fam), !is.na(GID)) %>%
    droplevels()
  
  n_obs <- nrow(data_subset)
  n_geno <- length(unique(data_subset$GID))
  n_fam <- length(unique(data_subset$Fam))
  
  cat("Observations:", n_obs, "\n")
  cat("Genotypes:", n_geno, "\n")
  cat("Families:", n_fam, "\n")
  
  if (n_obs < 10 || n_geno < 5) {
    cat("Insufficient data\n")
    return(NULL)
  }
  
  # Match kinship to phenotype data
  indp <- data_subset$GID %in% rownames(A_matrix)
  data_subset <- data_subset[indp, ]
  
  indg <- rownames(A_matrix) %in% data_subset$GID
  A_sub <- A_matrix[indg, indg]
  
  cat("After matching: n =", nrow(data_subset), ", genotypes =", nrow(A_sub), "\n")
  
  if (nrow(data_subset) < 10) {
    cat("Insufficient data after matching\n")
    return(NULL)
  }
  
  # Prepare genomic relationship matrix
  Ghat.bend <- G.tuneup(G = A_sub, bend = TRUE, eig.tol = 1e-03)$Gb
  Ginv.sparse <- G.inverse(G = Ghat.bend, sparse = TRUE)$Ginv
  
  # Verify matching
  match_result <- ASRgenomics::match.kinship2pheno(
    K = A_sub, 
    pheno.data = data_subset, 
    indiv = "GID"
  )
  
  # Ensure GID is factor
  data_subset$GID <- factor(data_subset$GID)
  data_subset$Fam <- factor(data_subset$Fam)
  
  # Center trait (important for genomic models)
  trait_mean <- mean(data_subset[[trait]], na.rm = TRUE)
  data_subset[[trait]] <- data_subset[[trait]] - trait_mean
  
  # Build formula
  formula_str <- paste0(trait, " ~ 1")
  
  # Fit model
  mod <- tryCatch({
    asreml(
      fixed = as.formula(formula_str),
      random = ~ vm(GID, Ginv.sparse) + Fam,
      residual = ~ units,
      data = data_subset,
      ai.sing = TRUE,
      trace = FALSE,
      maxit = 50
    )
  }, error = function(e) {
    cat("Model error:", e$message, "\n")
    return(NULL)
  })
  
  if (is.null(mod)) {
    return(NULL)
  }
  
  if (!mod$converge) {
    cat("Warning: Model did not converge\n")
  }
  
  # Extract variance components
  vc <- summary(mod)$varcomp
  cat("\nVariance Components:\n")
  print(vc)
  
  # Extract components
  
  Vg <- vc[1, "component"]      # Additive genetic variance
  Vfam <- vc[2, "component"]    # Family variance
  Ve <- vc[3, "component"]      # Residual variance
  
  # Mean "replication" (usually 1 for MQ traits)
  rep_counts <- table(data_subset$GID)
  r <- n_obs/n_geno
  
  # Total phenotypic variance
  Vp <- Vg + Vfam + Ve
  
  # Narrow-sense heritability (additive genetic only)
  h2 <- Vg / (Vg + Vfam + Ve/r)
  
  # Broad-sense heritability (genetic + family)
  H2 <- (Vg + Vfam) / (Vg + Vfam + Ve/r)
  
  # Proportion of variance explained
  prop_Vg <- Vg / Vp
  prop_Vfam <- Vfam / Vp
  prop_Ve <- Ve / Vp
  
  # Results
  results <- data.frame(
    Model = "Single_Env_MQ",
    Env = env,
    Trait = trait,
    Vg = round(Vg, 4),
    Vfam = round(Vfam, 4),
    Ve = round(Ve, 4),
    Vp = round(Vp, 4),
    prop_Vg = round(prop_Vg, 4),
    prop_Vfam = round(prop_Vfam, 4),
    prop_Ve = round(prop_Ve, 4),
    h2 = round(h2, 4),
    H2 = round(H2, 4),
    n_obs = n_obs,
    n_geno = n_geno,
    n_fam = n_fam,
    mean_rep = round(r, 2),
    converged = mod$converge,
    stringsAsFactors = FALSE
  )
  
  cat(sprintf("\n✓ h² (narrow-sense): %.4f\n", h2))
  cat(sprintf("✓ H² (broad-sense):  %.4f\n", H2))
  
  return(list(model = mod, results = results, varcomp = vc))
}

calculate_h2_mq_multi <- function(data, trait, A_matrix) {
  
  cat("\n========================================\n")
  cat("MULTI-ENVIRONMENT MODEL (MQ)\n")
  cat("Trait:", trait, "\n")
  cat("========================================\n\n")
  
  # Filter data
  data_clean <- data %>%
    filter(!is.na(!!sym(trait))) %>%
    filter(!!sym(trait) > 0) %>%
    filter(!is.na(Fam), !is.na(GID)) %>%
    droplevels()
  
  n_env <- length(unique(data_clean$Env))
  n_obs <- nrow(data_clean)
  n_geno <- length(unique(data_clean$GID))
  
  cat("Environments:", n_env, "\n")
  cat("Observations:", n_obs, "\n")
  cat("Genotypes:", n_geno, "\n\n")
  
  if (n_env < 2) {
    cat("Need at least 2 environments\n")
    return(NULL)
  }
  
  # Match kinship to phenotype
  indp <- data_clean$GID %in% rownames(A_matrix)
  data_clean <- data_clean[indp, ]
  
  indg <- rownames(A_matrix) %in% data_clean$GID
  A_sub <- A_matrix[indg, indg]
  
  cat("After matching: n =", nrow(data_clean), "\n\n")
  
  if (nrow(data_clean) < 20) {
    cat("Insufficient data after matching\n")
    return(NULL)
  }
  
  # Prepare genomic matrix
  Ghat.bend <- G.tuneup(G = A_sub, bend = TRUE, eig.tol = 1e-03)$Gb
  Ginv.sparse <- G.inverse(G = Ghat.bend, sparse = TRUE)$Ginv
  
  # Match and arrange
  ASRgenomics::match.kinship2pheno(K = A_sub, pheno.data = data_clean, indiv = "GID")
  data_clean <- data_clean %>% arrange(Env)
  
  data_clean$GID <- factor(data_clean$GID)
  data_clean$Fam <- factor(data_clean$Fam)
  data_clean$Env <- factor(data_clean$Env)
  
  # Center trait within each environment
  data_clean <- data_clean %>%
    group_by(Env) %>%
    mutate(!!sym(trait) := !!sym(trait) - mean(!!sym(trait), na.rm = TRUE)) %>%
    ungroup()
  
  # Build formula
  formula_str <- paste0(trait, " ~ Env")
  
  # Fit model
  cat("Fitting multi-environment model...\n")
  mod <- tryCatch({
    asreml(
      fixed = as.formula(formula_str),
      random = ~ vm(GID, Ginv.sparse) + 
                 GID:Env + 
                 Fam,
      residual = ~ dsum(~ idv(units) | Env),
      data = data_clean,
      ai.sing = TRUE,
      trace = FALSE,
      maxit = 100
    )
  }, error = function(e) {
    cat("Full model failed:", e$message, "\n")
    cat("Trying simplified model...\n")
    
    # Simplified model without G×E
    asreml(
      fixed = as.formula(formula_str),
      random = ~ vm(GID, Ginv.sparse) + Fam,
      residual = ~ dsum(~ idv(units) | Env),
      data = data_clean,
      ai.sing = TRUE,
      trace = FALSE,
      maxit = 100
    )
  })
  
  if (is.null(mod)) {
    return(NULL)
  }
  
  if (!mod$converge) {
    warning("Model did not converge fully")
  } else {
    cat("✓ Model converged\n\n")
  }
  
  # Extract variance components
  vc <- summary(mod)$varcomp
  print(vc)
  
  # Extract components
  Vg <- vc[grep("vm\\(GID", rownames(vc))[1], "component"]
  
  Vge_rows <- grep("GID:Env", rownames(vc))
  Vge <- if(length(Vge_rows) > 0) vc[Vge_rows[1], "component"] else 0
  
  Vfam <- vc[grep("^Fam", rownames(vc))[1], "component"]
  
  resid_rows <- grep("Env_|units", rownames(vc))
  Ve <- mean(vc[resid_rows, "component"])
  
  # Calculate metrics
  Vp <- Vg + Vge + Vfam + Ve
  
  prop_Vg <- Vg / Vp
  prop_Vge <- Vge / Vp
  prop_Vfam <- Vfam / Vp
  prop_Ve <- Ve / Vp
  
  # Mean replication
  rep_counts <- data_clean %>%
    group_by(Env, GID) %>%
    summarise(n = n(), .groups = 'drop')
  r <- n_obs/n_geno
  
  # Narrow-sense heritability
  h2 <- Vg / (Vg + Vge/n_env + Vfam + Ve/r)
  
  # Broad-sense heritability
  H2 <- (Vg + Vfam) / (Vg + Vge/n_env + Vfam + Ve/r)
  
  # Genetic correlation across environments
  rg_env <- Vg / (Vg + Vge)
  
  # Results
  results <- data.frame(
    Model = "Multi_Env_MQ",
    Trait = trait,
    Vg = round(Vg, 4),
    Vge = round(Vge, 4),
    Vfam = round(Vfam, 4),
    Ve = round(Ve, 4),
    Vp = round(Vp, 4),
    prop_Vg = round(prop_Vg, 4),
    prop_Vge = round(prop_Vge, 4),
    prop_Vfam = round(prop_Vfam, 4),
    prop_Ve = round(prop_Ve, 4),
    h2 = round(h2, 4),
    H2 = round(H2, 4),
    rg_env = round(rg_env, 4),
    n_env = n_env,
    n_obs = n_obs,
    n_geno = n_geno,
    mean_rep = round(r, 2),
    converged = mod$converge,
    stringsAsFactors = FALSE
  )
  
  cat("\n===== VARIANCE COMPONENTS =====\n")
  cat(sprintf("Genetic (Vg):         %8.4f  (%5.1f%%)\n", Vg, prop_Vg*100))
  cat(sprintf("G × Env (Vge):        %8.4f  (%5.1f%%)\n", Vge, prop_Vge*100))
  cat(sprintf("Family (Vfam):        %8.4f  (%5.1f%%)\n", Vfam, prop_Vfam*100))
  cat(sprintf("Error (Ve):           %8.4f  (%5.1f%%)\n", Ve, prop_Ve*100))
  cat(sprintf("\nh² (narrow-sense):    %.4f\n", h2))
  cat(sprintf("H² (broad-sense):     %.4f\n", H2))
  cat(sprintf("rg (environments):    %.4f\n", rg_env))
  
  return(list(model = mod, results = results, varcomp = vc))
}

calculate_h2_narrow_multi <- function(data, trait, A_matrix) {
  
  cat("\n========================================\n")
  cat("MULTI-ENVIRONMENT MODEL (Narrow-Sense)\n")
  cat("Trait:", trait, "\n")
  cat("========================================\n\n")
  
  # Filter data
  data_clean <- data %>%
    filter(!is.na(!!sym(trait))) %>%
    droplevels()
  
  n_env <- length(unique(data_clean$Env))
  n_obs <- nrow(data_clean)
  n_geno <- length(unique(data_clean$GID))
  
  cat("Environments:", n_env, "\n")
  cat("Observations:", n_obs, "\n")
  cat("Genotypes:", n_geno, "\n\n")
  
  if (n_env < 2) {
    cat("Need at least 2 environments\n")
    return(NULL)
  }
  
  # Match kinship to phenotype
  indp <- data_clean$GID %in% rownames(A_matrix)
  data_clean <- data_clean[indp, ]
  
  indg <- rownames(A_matrix) %in% data_clean$GID
  A_sub <- A_matrix[indg, indg]
  
  cat("After matching: n =", nrow(data_clean), "\n\n")
  
  if (nrow(data_clean) < 30) {
    cat("Insufficient data after matching\n")
    return(NULL)
  }
  
  # Prepare genomic matrix
  Ghat.bend <- G.tuneup(G = A_sub, bend = TRUE, eig.tol = 1e-03)$Gb
  Ginv.sparse <- G.inverse(G = Ghat.bend, sparse = TRUE)$Ginv
  
  # Match and arrange
  ASRgenomics::match.kinship2pheno(K = A_sub, pheno.data = data_clean, indiv = "GID")
  data_clean <- data_clean %>% arrange(Env, Row, Column)
  
  data_clean$GID <- factor(data_clean$GID)
  data_clean$Env <- factor(data_clean$Env)
  
  # Build formula
  formula_str <- paste0(trait, " ~ Env")
  
  # Fit model
  cat("Fitting multi-environment model...\n")
  mod <- tryCatch({
    asreml(
      fixed = as.formula(formula_str),
      random = ~ vm(GID, Ginv.sparse) + 
                 GID:Env + 
                 at(Env):ar1(Row):ar1(Column),
      residual = ~ dsum(~ idv(units) | Env),
      data = data_clean,
      trace = FALSE,
      maxit = 100
    )
  }, error = function(e) {
    cat("Full model failed:", e$message, "\n")
    cat("Trying simplified model...\n")
    
    # Simplified model without G×E
    asreml(
      fixed = as.formula(formula_str),
      random = ~ vm(GID, Ginv.sparse) + 
                 at(Env):ar1(Row):ar1(Column),
      residual = ~ dsum(~ idv(units) | Env),
      data = data_clean,
      trace = FALSE,
      maxit = 100
    )
  })
  
  if (is.null(mod)) {
    return(NULL)
  }
  
  if (!mod$converge) {
    warning("Model did not converge fully")
  } else {
    cat("✓ Model converged\n\n")
  }
  
  # Extract variance components
  vc <- summary(mod)$varcomp
  print(vc)
  
  # Extract components
  Vg <- vc[grep("vm\\(GID", rownames(vc))[1], "component"]
  
  Vge_rows <- grep("GID:Env", rownames(vc))
  Vge <- if(length(Vge_rows) > 0) vc[Vge_rows[1], "component"] else 0
  
  spatial_rows <- grep("Row.*Column|ar1.*ar1", rownames(vc))
  Vspatial <- if(length(spatial_rows) > 0) mean(vc[spatial_rows, "component"]) else 0
  
  resid_rows <- grep("Env_|units", rownames(vc))
  Ve <- mean(vc[resid_rows, "component"])
  
  # Calculate metrics
  Vp <- Vg + Vge + Vspatial + Ve
  
  prop_Vg <- Vg / Vp
  prop_Vge <- Vge / Vp
  prop_Vspatial <- Vspatial / Vp
  prop_Ve <- Ve / Vp
  
  # Plot basis
  h2_plot <- Vg / Vp
  
  # Mean replication
  rep_counts <- data_clean %>%
    group_by(Env, GID) %>%
    summarise(n = n(), .groups = 'drop')
  r <- n_obs/n_geno
  
  # Entry-mean basis (narrow-sense heritability)
  h2_mean <- Vg / (Vg + Vge/n_env + Vspatial/r + Ve/r)
  
  # Genetic correlation across environments (stability)
  rg_env <- Vg / (Vg + Vge)
  
  # Accuracy
  accuracy <- sqrt(h2_mean)
  
  # Results
  results <- data.frame(
    Model = "Narrow_Sense_Multi",
    Trait = trait,
    Vg = round(Vg, 4),
    Vge = round(Vge, 4),
    Vspatial = round(Vspatial, 4),
    Ve = round(Ve, 4),
    Vp = round(Vp, 4),
    prop_Vg = round(prop_Vg, 4),
    prop_Vge = round(prop_Vge, 4),
    prop_Vspatial = round(prop_Vspatial, 4),
    prop_Ve = round(prop_Ve, 4),
    h2_plot = round(h2_plot, 4),
    h2_mean = round(h2_mean, 4),
    rg_env = round(rg_env, 4),
    accuracy = round(accuracy, 4),
    n_env = n_env,
    n_obs = n_obs,
    n_geno = n_geno,
    mean_rep = round(r, 2),
    converged = mod$converge,
    stringsAsFactors = FALSE
  )
  
  cat("\n===== VARIANCE COMPONENTS =====\n")
  cat(sprintf("Additive Genetic (Vg):    %8.4f  (%5.1f%%)\n", Vg, prop_Vg*100))
  cat(sprintf("G × Env (Vge):            %8.4f  (%5.1f%%)\n", Vge, prop_Vge*100))
  cat(sprintf("Spatial:                  %8.4f  (%5.1f%%)\n", Vspatial, prop_Vspatial*100))
  cat(sprintf("Error (Ve):               %8.4f  (%5.1f%%)\n", Ve, prop_Ve*100))
  cat(sprintf("\nh² (narrow-sense, mean):  %.4f\n", h2_mean))
  cat(sprintf("Accuracy:                 %.4f\n", accuracy))
  cat(sprintf("rg (environments):        %.4f\n", rg_env))
  
  return(list(model = mod, results = results, varcomp = vc))
}

calculate_h2_single_env <- function(data, env, trait) {

  data_subset <- data %>%
    filter(Env == env) %>%

    arrange(Row, Column)
  
  # Check data quality
  n_obs <- nrow(data_subset)
  n_geno <- length(unique(data_subset$GID))
  
  if (n_obs < 20 || n_geno < 5) {
    cat("Insufficient data (n =", n_obs, ", genotypes =", n_geno, ")\n")
    return(NULL)
  }
  
  # Build formula
  formula_str <- paste0(trait, " ~ 1")
  
  # Fit model
  mod <- tryCatch({
    asreml(
      fixed = as.formula(formula_str),
      random = ~ GID + ar1(Row):ar1(Column),
      residual = ~ idv(units),
      data = data_subset,
      trace = FALSE,
      maxit = 50
    )
  }, error = function(e) {

  })
  
  if (is.null(mod)) {

  if (!mod$converge) {

  }

  vc <- summary(mod)$varcomp

  Vg <- vc[grep("^GID", rownames(vc))[1], "component"]

  Vp <- Vg + Vrc + Ve
  h2_plot <- Vg / Vp

  # Cullis heritability
  h2_cullis <- tryCatch({
    pev <- mod$vcoeff$GID
    if (!is.null(pev)) 1 - mean(pev) / (2 * Vg) else NA
  }, error = function(e) NA)
  
  # Results
  results <- data.frame(
    Model = "Single_Env",
    Env = env,
    Trait = trait,
    Vg = round(Vg, 4),
    Vrc = round(Vrc, 4),
    Ve = round(Ve, 4),

    n_obs = n_obs,
    n_geno = n_geno,
    mean_rep = round(r, 2),
    converged = mod$converge,
    stringsAsFactors = FALSE
  )
  
  cat(sprintf("\n✓ h² (mean basis): %.4f\n", h2_mean))
  
  return(list(model = mod, results = results, varcomp = vc))
}

model_1_single <- function(data, env, timepoint, vi) {
  
  cat("\n========================================\n")
  cat("MODEL 1: Single Env-Timepoint\n")
  cat("Env:", env, "| Timepoint:", timepoint, "| VI:", vi, "\n")
  cat("========================================\n\n")
  
  # Filter and prepare data
  data_subset <- data %>%
    filter(Env == env, timepoint == !!timepoint) %>%  # FIXED: Added timepoint filter
    filter(!is.na(!!sym(vi))) %>%
    mutate(
      GID = factor(GID),
      Row = factor(Row),
      Column = factor(Column),
      plot_id = factor(plot_id)
    ) %>%
    arrange(Row, Column)  # Sort for spatial model
  
  # Check data
  n_obs <- nrow(data_subset)
  n_geno <- length(unique(data_subset$GID))
  
  if (n_obs < 20 || n_geno < 5) {
    cat("Insufficient data (n_obs =", n_obs, ", n_geno =", n_geno, ")\n")
    return(NULL)
  }
  
  # Build formula
  formula_str <- paste0(vi, " ~ 1")
  
  # Fit model
  mod <- tryCatch({
    asreml(
      fixed = as.formula(formula_str),
      random = ~ GID + ar1(Row):ar1(Column),
      residual = ~ idv(units),
      data = data_subset,
      trace = FALSE,
      maxit = 50
    )
  }, error = function(e) {
    cat("Model error:", e$message, "\n")
    return(NULL)
  })
  
  if (is.null(mod)) {
    return(NULL)
  }
  
  if (!mod$converge) {
    cat("Warning: Model did not converge\n")
  }
  
  # Extract variance components
  vc <- summary(mod)$varcomp
  cat("\nVariance Components:\n")
  print(vc)
  
  Vg <- vc[grep("^GID", rownames(vc))[1], "component"]
  Vrc_rows <- grep("Row.*Column", rownames(vc))
  Vrc <- if(length(Vrc_rows) > 0) vc[Vrc_rows[1], "component"] else 0
  Ve <- vc[grep("units|residual|R!", rownames(vc))[1], "component"]
  
  # Calculate heritability
  rep_counts <- table(data_subset$GID)
  r <- n_obs/n_geno
  
  Vp <- Vg + Vrc + Ve
  h2_plot <- Vg / Vp
  h2_mean <- Vg / (Vg + Vrc/r + Ve/r)
  
  # Cullis heritability
  h2_cullis <- tryCatch({
    pev <- mod$vcoeff$GID
    if (!is.null(pev)) 1 - mean(pev) / (2 * Vg) else NA
  }, error = function(e) NA)
  
  results <- data.frame(
    Model = "Model_1_Single",
    Env = env,
    Timepoint = timepoint,
    VI = vi,
    Vg = round(Vg, 6),
    Vrc = round(Vrc, 6),
    Ve = round(Ve, 6),
    Vp = round(Vp, 6),
    h2_plot = round(h2_plot, 4),
    h2_mean = round(h2_mean, 4),
    h2_cullis = round(h2_cullis, 4),
    n_obs = n_obs,
    n_geno = n_geno,
    mean_rep = round(r, 2),
    converged = mod$converge,
    stringsAsFactors = FALSE
  )
  
  cat(sprintf("\n✓ h² (plot): %.4f | h² (mean): %.4f | h² (Cullis): %.4f\n", 
              h2_plot, h2_mean, h2_cullis))
  
  return(list(model = mod, results = results, varcomp = vc))
}
