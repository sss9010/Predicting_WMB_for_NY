  mutate(Date = as.Date(str_remove(as.character(Date), " UTC$")))

# attach sowing dates and compute days-from-sowing (JDD) and from Jan 1 (JDD1)
JD_map <- flight_d %>%
  left_join(sow_map, by = "Year") %>%
  mutate(
    JDD  = as.integer(Date - sd),
    JDD1 = as.integer(Date - as.Date(paste0(Year, "-01-01")))
  )

# weather → daily mean temp and cumulative GDD per year
base_temp <- 40
weather <- readr::read_csv("data/weather_dataq_ithaca.csv", show_col_types = FALSE) %>%
  mutate(
    DATE    = as.Date(DATE),
    year    = year(DATE),
    t_avg   = (TMAX + TMIN) / 2,
    gdd_day = pmax(t_avg - base_temp, 0)
  ) %>%
  arrange(year, DATE) %>%
  group_by(year) %>%
  mutate(GDD_cum = cumsum(gdd_day)) %>%
  ungroup()

# GDD on flight dates: join cumulative GDD by exact date & year
JD_map <- JD_map %>%
  left_join(weather %>% transmute(Year = year, Date = DATE, GDD = GDD_cum),
            by = c("Year","Date"))

# average HD_JD per year (use Env→Year map so it matches sow_map)
year_map <- sow_map %>% dplyr::select(Env = Environment, Year)
data_ag  <- data_ag %>% mutate(Env = as.character(Env)) %>% left_join(year_map, by = "Env")

avg_HD_by_year <- data_ag %>%
  group_by(Year) %>%
  summarise(mean_HD_JD = mean(HD_JD, na.rm = TRUE), .groups = "drop") %>%
  mutate(JD_date = as.Date(paste0(Year, "-01-01")) + mean_HD_JD)

# GDD at the mean heading date (vectorized join to weather)
avg_HD_by_year <- avg_HD_by_year %>%
  left_join(weather %>% transmute(Year = year, JD_date = DATE, GDD_JD = GDD_cum),
            by = c("Year","JD_date"))

# merge heading-date info to flight map and compute distance from HD
JD_map <- JD_map %>%
  left_join(avg_HD_by_year, by = "Year") %>%

# 2) safe correlation helper
safe_cor <- function(x, y, method = "pearson") {
  ok <- is.finite(x) & is.finite(y) & !is.na(x) & !is.na(y)
  if (sum(ok) < 2L) return(NA_real_)
  if (sd(x[ok]) == 0 || sd(y[ok]) == 0) return(NA_real_)
  suppressWarnings(cor(x[ok], y[ok], method = method))
}
