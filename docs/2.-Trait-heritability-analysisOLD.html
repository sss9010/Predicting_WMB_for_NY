model_list <- list()
ratio_list <- list()

# Loop over each trait
for (tr in traits) {
  
  # Nested list/vector for storing results across environments
  model_env  <- list()
  ratio_env  <- numeric(length(envs)) 
  # Loop over each environment
  for (i in seq_along(envs)) {
    e <- envs[i]
    # Create a temporary subset for environment e
    data_sub <- data %>% 
      dplyr::filter(Env == e)
    data_sub$GID<- as.factor(data_sub$GID)
    # Center the trait by subtracting its mean (within env)
    data_sub[[tr]] <- data_sub[[tr]] - mean(data_sub[[tr]], na.rm = TRUE)
    # Fit the model
    mod_M1 <- tryCatch({
      asreml(
        fixed    = as.formula(paste(tr, "~  1")),
        random   = ~ GID  +  ~ar1(Row):ar1(Column),,
        residual = ~ units,
        data     = data_sub   )
    }, error = function(err) {
      
      message(paste("Error fitting trait:", tr, "in environment:", e))
      message("Error message:", err$message)
      return(NA)
    })
    # Store the model in a nested list
    model_env[[e]] <- mod_M1
    plot(mod_M1)
    # If successful, compute the variance ratio
    if (inherits(mod_M1, "asreml")) {
      vc <- summary(mod_M1)$varcomp
      rep_counts <- table(data_sub$GID)
      r <- mean(rep_counts)
      # Assuming first row is genetic and second row is residual
      ratio_env[i] <- vc[1, 1] / (vc[1, 1]  + vc[2, 1]+ vc[5, 1]/r)

    } else {
      ratio_env[i] <- NA
    }
  }
  model_list[[tr]] <- model_env
  ratio_list[[tr]] <- ratio_env
}

ratio_rows <- list()

for (tr in traits) {
  model_env <- list()
  
  for (e in envs) {
    data_sub <- data %>%
      filter(Env == e) %>%
      filter(.data[[tr]] > 0)
    data_sub<- data_sub %>% filter(!Fam == "NA") %>% filter(!GID == "NA")
    data_sub$Fam<- as.factor(data_sub$Fam)
    # Preprocessing as before...
    load("data/geno/WMB_master_geno.RData")
    indp = data_sub$GID %in% rownames(A1)
    data_sub = data_sub[indp,]
    indg = rownames(A1) %in% data_sub$GID
    A1 = A1[indg, indg]
    Ghat.bend <- G.tuneup(G = A1, bend = TRUE, eig.tol = 1e-03)$Gb
    Ginv.sparse <- G.inverse(G = Ghat.bend, sparse = TRUE)$Ginv
    ASRgenomics::match.kinship2pheno(K = A1, pheno.data = data_sub, indiv = "GID")
    data_sub$GID <- as.factor(data_sub$GID)
    
    # Center trait
    data_sub[[tr]] <- data_sub[[tr]] - mean(data_sub[[tr]], na.rm = TRUE)
    
    mod_M1 <- tryCatch({
      asreml(
        fixed    = as.formula(paste(tr, "~ 1")),
        random   = ~ vm(GID, Ginv.sparse) + ~Fam,
        residual = ~ units,
        data     = data_sub,
        ai.sing  = TRUE
      )
    }, error = function(err) {
      message(paste("Error fitting trait:", tr, "in environment:", e))
      return(NA)
    })
    
    model_env[[e]] <- mod_M1
    
    if (inherits(mod_M1, "asreml")) {
      vc <- summary(mod_M1)$varcomp
      r <- mean(table(data_sub$GID))
      H2 <- (vc[1, 1] + + vc[2, 1]) / (vc[1, 1] + vc[2, 1] + vc[3, 1] / r)
      h2 <- vc[1, 1]  / (vc[1, 1] + vc[2, 1] + vc[3, 1] / r)
      
      ratio_rows[[length(ratio_rows) + 1]] <- data.frame(
        Trait = tr,
        Environment = e,
        H2 = H2,
        h2 = h2
      )
    }
  }
  
  model_list[[tr]] <- model_env
}

## ------------------------------------------------------------------
## outer loop : traits
## ------------------------------------------------------------------
for (tr in traits) {

  model_env  <- list()
  ratio_env  <- list()

  ## ---------------------------------------------------------------
  ## middle loop : environments
  ## ---------------------------------------------------------------
  for (e in envs) {

    data_env <- data %>% 
      filter(Env == e, .data[[tr]] > 0)           # keep only positive values

    ## nothing to do if env has no data for this trait
    if (nrow(data_env) == 0) next

    ## time‑points present in this env × trait subset
    tps <- sort(unique(data_env$timepoint))
    

    model_tp <- list()
    ratio_tp <- numeric(length(tps))

    ## ------------------------------------------------------------
    ## inner loop : time‑points
    ## ------------------------------------------------------------
    for (j in seq_along(tps)) {

      tp <- tps[j]

      data_sub <- data_env %>% 
        filter(timepoint == tp) %>% 
        mutate(
          GID      = factor(GID),
          !!tr     := .data[[tr]] - mean(.data[[tr]], na.rm = TRUE)  # centre
        )

      mod_M1 <- tryCatch(
        asreml(
          fixed    = reformulate("1", response = tr),
          random   = ~GID +  ~ar1(Row):ar1(Column),
          residual = ~ units,
          data     = data_sub,
          ai.sing  = TRUE
        ),
        error = function(err) {
          message("Error fitting trait ", tr,
                  " | env ", e, " | tp ", tp, "\n  ",
                  err$message)
          NA
        }
      )

      model_tp[[as.character(tp)]] <- mod_M1

      ## --------- variance ratio if model converged ---------------
      if (inherits(mod_M1, "asreml")) {
        vc          <- summary(mod_M1)$varcomp
        rep_counts  <- table(data_sub$GID)
        r           <- mean(rep_counts)
        ratio_tp[j] <- vc[1, 1] / (vc[1, 1] + vc[5, 1] / r)
      } else {
        ratio_tp[j] <- NA
      }
    } # end time‑point loop

    names(model_tp) <- names(ratio_tp) <- as.character(tps)
    model_env [[e]] <- model_tp
    ratio_env [[e]] <- ratio_tp
  } # end environment loop

  model_list[[tr]] <- model_env
  ratio_list[[tr]] <- ratio_env

envs   <- unique(df_ag$Env)[c(1,3,4)]
data<- data_ag 

model_list <- list()
ratio_list <- list()

# Loop over each trait
for (tr in traits) {
  
  # Nested list/vector for storing results across environments
  model_env  <- list()
  ratio_env  <- numeric(length(envs)) 
  
  # Loop over each environment
  for (i in seq_along(envs)) {
    e <- envs[i]
    
    # Create a temporary subset for environment e
    data_sub <- data %>% 
      dplyr::filter(Env == e)
    data_sub$GID<- as.factor(data_sub$GID)

    
    # Center the trait by subtracting its mean (within env)
    data_sub[[tr]] <- data_sub[[tr]] - mean(data_sub[[tr]], na.rm = TRUE)
    
    # Fit the model
    mod_M1 <- tryCatch({
      
      asreml(
        fixed    = as.formula(paste(tr, "~ 1")),
        random   = ~ GID ,#+ ~ar1(Row):ar1(Column),
        residual = ~ units,
        data     = data_sub,
        ai.sing  = TRUE   # Helps diagnose singularities
      )
      
      
    }, error = function(err) {
      
      message(paste("Error fitting trait:", tr, "in environment:", e))
      message("Error message:", err$message)
      return(NA)
    })
    
    # Store the model in a nested list
    model_env[[e]] <- mod_M1
    
    # If successful, compute the variance ratio
    if (inherits(mod_M1, "asreml")) {
      vc <- summary(mod_M1)$varcomp
      rep_counts <- table(data_sub$GID)
      r <- mean(rep_counts)
      print(r)
      # Assuming first row is genetic and second row is residual
      ratio_env[i] <- vc[1, 1] / (vc[1, 1] + vc[2, 1]/r)
    } else {
      ratio_env[i] <- NA
    }
  }
  
  # Store all environment-specific models for this trait
  model_list[[tr]] <- model_env
  # Store the vector of ratio values across environments
  ratio_list[[tr]] <- ratio_env
}

# Create a data frame from the ratio_list
ratio_df <- data.frame(
  Trait       = rep(traits, each = length(envs)),
  Environment = rep(envs, times = length(traits)),
  Ratio       = unlist(ratio_list),
  stringsAsFactors = FALSE
)
