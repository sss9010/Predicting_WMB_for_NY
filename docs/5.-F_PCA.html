# Loop over environments and VIs
for (env in names(VImats)) {
  mats <- names(VImats[[env]])[-10]

  for (vi in mats) {
    df <- VImats[[env]][[vi]] %>% na.omit()
    rownames(df) <- df$plot_id
    mat <- df[, -1]
    mat_scaled <- scale(mat)

    ## --- PCA ---
    pca <- prcomp(mat_scaled, center = FALSE, scale. = FALSE)

    # Variance explained
    pca_var <- pca$sdev^2
    pca_var_df <- data.frame(
      Env = env, VI = vi,
      PC = paste0("PC", seq_along(pca_var)),
      Variance = pca_var / sum(pca_var)
    )
    PCA_variance_df <- bind_rows(PCA_variance_df, pca_var_df)

    # Loadings
    loadings <- as.data.frame(pca$rotation)
    #loadings$Timepoint <- as.numeric(gsub("[^0-9]", "", colnames(mat)))
    loadings$Timepoint <- as.numeric(colnames(mat))
    loadings$Env <- env
    loadings$VI <- vi
    PCA_loadings_df <- bind_rows(
      PCA_loadings_df,
      loadings %>%
        pivot_longer(cols = starts_with("PC"), names_to = "PC", values_to = "Loading")
    )

    # Scores
    scores <- as.data.frame(pca$x)
    scores$plot_id <- df$plot_id
    scores$Env <- env
    scores$VI <- vi
    PCA_scores_df <- bind_rows(PCA_scores_df, scores)

    ## --- FPCA ---
    long_df <- df %>%
      pivot_longer(-plot_id, names_to = "Time", values_to = "value")
    long_df$Time<- as.numeric(long_df$Time)

    Ly <- split(long_df$value, long_df$plot_id)
    Lt <- split(long_df$Time, long_df$plot_id)

    fpca <- FPCA(Ly = Ly, Lt = Lt, optns = list(dataType = 'Sparse', methodSelectK = 3))

    # Eigenfunctions
    eigen_df_tmp <- data.frame(
      Env = env, VI = vi,
      Time = fpca$workGrid,
      FPC1 = fpca$phi[, 1],
      FPC2 = fpca$phi[, 2],
      FPC3 = fpca$phi[, 3]
    )
    FPCA_eigen_df <- bind_rows(FPCA_eigen_df, eigen_df_tmp)

    # FPCA scores
    scores_fpca <- as.data.frame(fpca$xiEst)
    colnames(scores_fpca) <- paste0("FPC", 1:ncol(scores_fpca))
    scores_fpca$plot_id <- df$plot_id
    scores_fpca$Env <- env
    scores_fpca$VI <- vi
    FPCA_scores_df <- bind_rows(FPCA_scores_df, scores_fpca)
  }
}

FPCA_var_df <- FPCA_scores_df %>%
  #select(starts_with("FPC"), Env, VI) %>%  # Keep Env and VI for grouping
  pivot_longer(cols = starts_with("FPC"), names_to = "FPC", values_to = "score") %>%
  group_by(FPC, Env, VI) %>%
  summarise(variance = var(score), .groups = "drop") %>%
  group_by(Env, VI) %>%
  mutate(
    variance_ratio = variance / sum(variance),
    FPC = factor(FPC, levels = unique(FPC))
  )

# 1) Helper to standardize Env
std_env <- function(x) {
  x %>%
    as.character() %>%
    toupper() %>%
    str_replace_all("\\s+", "") %>%     
    str_replace("^(\\d{2})([A-Z]+)$", "\\2\\1") %>%
    str_replace("^([A-Z]+)(\\d{2})$", "\\1\\2")
}
