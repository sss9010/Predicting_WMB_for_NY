# Suppose we have columns: GID, TP, NDVI, EVI, GNDVI
# We'll pivot so each row = GID, each column = (TP+VI)

df_wide <- df %>%
  select(GID, TP, NDVI, EVI, GNDVI) %>%
  pivot_longer(cols = c(NDVI, EVI, GNDVI), 
               names_to = "VIType", values_to = "VIValue") %>%
  unite("Feature", c(TP, VIType), remove = TRUE) %>%  
  # => Feature might look like "TP1_NDVI", "TP1_EVI", ...
  pivot_wider(names_from = Feature, values_from = VIValue)

# Now each row = GID, columns = all combinations of TP and VI
# If some data are missing, consider imputing or setting NA.

# Optionally average replicate observations of the same GID
df_wide <- df_wide %>%
  group_by(GID) %>%
  summarize_all(mean, na.rm = TRUE) %>%
  ungroup()

# Convert to matrix X. Rownames = GID, columns = features
X_wide <- as.matrix(df_wide[,-1])  # drop GID column
rownames(X_wide) <- df_wide$GID

# ---- Step 2: PCA on X_wide ----
