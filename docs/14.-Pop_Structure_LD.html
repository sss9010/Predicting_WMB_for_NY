1 Overview

This script characterises the population structure and linkage disequilibrium (LD) of the NY Winter Malting Barley (NY_WMB) marker data. The genotype input is:

File Format Description
LTT_geno_50K.csv Semicolon-delimited, markers × individuals AA / AB / BB calls, converted to 0 / 1 / 2 additive coding

SNP chromosome and physical position information is sourced from the Morex V3 assembly annotation.

Population structure is explored via:

  • Principal Component Analysis (PCA) on the centred marker matrix
  • Discriminant Analysis of Principal Components (DAPC)
  • Genomic kinship heatmap

Linkage disequilibrium is characterised via:

  • Genome-wide and per-chromosome LD decay (r² vs physical distance)
  • Per-chromosome LD heatmaps on a marker subsample
  • Estimation of the distance at which r² drops to half its maximum (LD half-life)

# AA = homozygous reference → 0

# anything else (e.g. "--", "NC", "") → NA
aabb_to_012 <- function(x) {
  x <- trimws(toupper(x))
  out <- rep(NA_real_, length(x))
  out[x == "AA"]              <- 0
  out[x %in% c("AB", "BA")] <- 1
  out[x == "BB"]              <- 2
  out
}

# ── Impute missing values with column mean before PCA ─────────────────────

  if (any(nas)) geno_imp[nas, j] <- col_means[j]

# ── Centre markers (subtract column mean) ─────────────────────────────────

# ── Fast r² function for two marker vectors ────────────────────────────────
r2_pair <- function(x, y) {
  ok   <- complete.cases(x, y)
  if (sum(ok) < 10) return(NA_real_)
  x <- x[ok]; y <- y[ok]
  # r² via correlation of allele counts
  cor(x, y)^2
}

# ── Sample pairs within a distance window and compute r² ──────────────────
compute_ld_decay <- function(chrom, n_pairs = 5000, max_dist_mb = 300) {
  chr_map  <- map_sub %>% filter(Chr == chrom) %>% arrange(Pos)
  if (nrow(chr_map) < 10) return(NULL)
  markers  <- chr_map$Marker
  G_chr    <- geno_mapped[, markers, drop = FALSE]
  n_m      <- length(markers)

  # Random pairs sampling
  n_sample <- min(n_pairs, n_m * (n_m - 1) / 2)
  pairs    <- matrix(sample.int(n_m, 2 * n_sample, replace = TRUE),
                     ncol = 2)
  pairs    <- pairs[pairs[,1] != pairs[,2], , drop = FALSE]
  pairs    <- pairs[1:min(n_sample, nrow(pairs)), , drop = FALSE]

  map_dfr(seq_len(nrow(pairs)), function(k) {
    i <- pairs[k, 1]; j <- pairs[k, 2]
    dist_mb <- abs(chr_map$Pos[i] - chr_map$Pos[j]) / 1e6
    if (dist_mb > max_dist_mb) return(NULL)
    r2 <- r2_pair(G_chr[, i], G_chr[, j])
    tibble(Chr = chrom, dist_mb = dist_mb, r2 = r2)
  })
}
