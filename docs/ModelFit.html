library(dplyr)
library(asreml)

# Suppose we want 5 folds within each environment.
set.seed(1234)
data_cv <- data %>%
  group_by(LocYear) %>%
  mutate(
    FoldID = sample(rep(1:5, length.out = n()))  # random assignment to 5 folds
  ) %>%
  ungroup()

# Data structure now has data_cv$FoldID with values 1..5 for each LocYear
folds <- unique(data_cv$FoldID)
res_kfold <- data.frame(Env = character(), Fold = integer(), Corr = numeric())

for(e in unique(data_cv$LocYear)) {
  for(f in folds) {
    
    # 1. Copy the data subset for this environment e
    env_data <- data_cv[data_cv$LocYear == e, ]
    
    # 2. Set trait = NA in fold f
    env_data$trait[env_data$FoldID == f] <- NA
    
    # 3. Fit the model (example: GBLUP only)
    Mod_kfold <- asreml(
      fixed    = trait ~ 1,
      random   = ~ vm(GID, Ginv.sparse),
      residual = ~ dsum(~ units),
      data     = env_data
    )
    
    # 4. Predict for the withheld fold f
    #    We can do a genotype-level prediction in environment e
    pred_vals <- predict(
      Mod_kfold, 
      classify = "GID",  
      # or "GID:LocYear" if needed, but here data is single-env now
      # since we subset by environment e
    )$pvals
    
    # 5. Merge with the actual data from fold f
    held_out <- data_cv[data_cv$LocYear == e & data_cv$FoldID == f, 
                        c("GID", "LocYear", "trait")]
    merged <- merge(held_out, pred_vals[, c("GID", "predicted.value")],
                    by = "GID")
    
    # 6. Compute correlation
    cor_val <- cor(merged$trait, merged$predicted.value, use = "complete.obs")
    
    # 7. Store results
    res_kfold <- rbind(res_kfold, 
                       data.frame(Env = e, Fold = f, Corr = cor_val))
  }
}

# Summarize results
res_kfold
res_kfold_summary <- res_kfold %>%
  group_by(Env) %>%
  summarize(MeanCorr = mean(Corr, na.rm = TRUE),
            SDCorr   = sd(Corr, na.rm = TRUE))
