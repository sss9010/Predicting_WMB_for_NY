# Convert a genotype matrix/data frame (cols = markers, rows = samples) to 0/1/2
code012_from_iupac <- function(G, minor_is_2 = TRUE) {
  G <- as.data.frame(G, stringsAsFactors = FALSE)

  # normalize calls to single-letter IUPAC (A,C,G,T,R,Y,S,W,K,M,N)
  norm_one <- function(x) {
    x <- toupper(trimws(as.character(x)))
    x[x %in% c("-", ".", "", "NA")] <- "N"
    x <- gsub("/", "", x); x <- gsub("\\|", "", x)
    # map two-letter diplotypes to IUPAC
    map2 <- c(AA="A", CC="C", GG="G", TT="T",
              AG="R", GA="R", CT="Y", TC="Y",
              CG="S", GC="S", AT="W", TA="W",
              GT="K", TG="K", AC="M", CA="M")
    two <- nchar(x) == 2
    x[two] <- map2[x[two]]
    # if already "AA","BB" style with same letter, collapse
    x[nchar(x) == 1 & !(x %in% c("A","C","G","T","R","Y","S","W","K","M","N"))] <- NA
    x  }
  IUPAC <- list(
    A = "A", C = "C", G = "G", T = "T",
    R = c("A","G"), Y = c("C","T"), S = c("G","C"),
    W = c("A","T"), K = c("G","T"), M = c("A","C")  )

  out <- matrix(NA_real_, nrow = nrow(G), ncol = ncol(G),
                dimnames = list(rownames(G), colnames(G)))

  for (j in seq_len(ncol(G))) {
    col <- norm_one(G[[j]])
    # infer the two alleles from data (weighted by hets)
    alleles_obs <- unlist(IUPAC[col[!is.na(col) & col != "N"]])
    alleles_obs <- alleles_obs[alleles_obs %in% c("A","C","G","T")]
    if (length(alleles_obs) == 0L) next
    af <- sort(table(alleles_obs), decreasing = TRUE)

    if (length(af) == 1L) {
      major <- names(af)[1]; minor <- setdiff(c("A","C","G","T"), major)[1]
    } else {
      major <- names(af)[1]; minor <- names(af)[2]    }

    out[, j] <- vapply(col, function(z) {
      if (is.na(z) || z == "N") return(NA_real_)
      if (z %in% c("A","C","G","T")) {
        if (z == major) return(0)
        if (z == minor) return(if (minor_is_2) 2 else 2)
        return(NA_real_) # third allele present -> set NA
      } else {        # heterozygote codes
        pair <- IUPAC[[z]]
        if (all(c(major, minor) %in% pair)) return(1)
        return(NA_real_)   }
    }, numeric(1))  }
  as.data.frame(out, check.names = FALSE)}

# Build per-Env wide matrices for *each* VI AND a "combined" matrix per Env.
envs <- unique(multi_f$Env)


# Helper to wide-pivot one VI in one Env
wide_one_vi <- function(df_env, vi_name) {
  df_env %>%
    dplyr::select(plot_id, Timepoint, all_of(vi_name)) %>%
    distinct(plot_id, Timepoint, .keep_all = TRUE) %>%
    tidyr::pivot_wider(names_from = Timepoint, values_from = all_of(vi_name))
}

# Helper to wide-pivot all VI columns together (“combined”)
wide_combined <- function(df_env) {
  df_env %>%
    dplyr::select(plot_id, Timepoint, all_of(VI_cols)) %>%
    distinct(plot_id, Timepoint, .keep_all = TRUE) %>%
    tidyr::pivot_wider(names_from = Timepoint, values_from = all_of(VI_cols))
}

# Construct VImats as: list[Env][[vi]] and [[combined]]
VImats <- setNames(vector("list", length(envs)), envs)

for (e in envs) {
  df_e <- multi_f %>% filter(Env == e)
  # per-VI
  per_vi <- setNames(vector("list", length(VI_cols)), VI_cols)
  for (vi in VI_cols) per_vi[[vi]] <- wide_one_vi(df_e, vi)
  # combined
  per_vi[["combined"]] <- wide_combined(df_e)
  VImats[[e]] <- per_vi
}
